        // 7. دالة الإملاء الصوتي (تحويل الكلام إلى نص) - ميزة مساندة للتنوع العصبي

        function startDictation(inputId, btn) {

            // التحقق من دعم المتصفح لميزة التعرف على الصوت

            const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

            if (!SpeechRecognition) {

                Swal.fire('عذراً!', 'متصفحك الحالي لا يدعم ميزة الإملاء الصوتي. يرجى استخدام متصفح Google Chrome.', 'error');

                return;

            }


            const recognition = new SpeechRecognition();

            recognition.lang = 'ar-SA'; // تعيين اللغة العربية

            recognition.interimResults = false; // ننتظر حتى ينهي الطفل كلامه

            recognition.maxAlternatives = 1;


            // عند بدء التسجيل: نضيف تأثير النبض الأحمر للزر ليعرف الطفل أنه يعمل

            recognition.onstart = function() {

                btn.classList.add('recording');

                btn.title = "أنا أستمع... تحدث الآن";

            };


            // عند التقاط الصوت وتحويله لنص

            recognition.onresult = function(event) {

                const transcript = event.results[0][0].transcript;

                const input = document.getElementById(inputId);

                

                // إذا كان هناك نص سابق، نضيف مسافة ثم النص الجديد، وإلا نضع النص مباشرة

                if(input.value.trim() !== '') {

                    input.value += ' ' + transcript;

                } else {

                    input.value = transcript;

                }

            };


            // عند انتهاء التسجيل أو حدوث خطأ نوقف التأثير البصري

            recognition.onend = function() {

                btn.classList.remove('recording');

                btn.title = "تحدث بدلاً من الكتابة";

            };


            recognition.onerror = function(event) {

                btn.classList.remove('recording');

                console.error("خطأ في التعرف على الصوت:", event.error);

                

                // إضافة معالجة ذكية لحظر المايكروفون من قبل Google Sites

                if (event.error === 'not-allowed') {

                    Swal.fire({

                        title: 'تنبيه أمان 🛡️',

                        text: 'نظام الأمان في هذا الموقع يمنع تشغيل المايكروفون من الداخل. لا تقلق يا بطل، يمكنك كتابة مشاعرك يدوياً في المربع!',

                        icon: 'info',

                        confirmButtonText: 'حسناً، سأكتب يدوياً',

                        confirmButtonColor: '#0ea5e9'

                    });

                }

            };


            // تشغيل المايكروفون

            try {

                recognition.start();

            } catch (err) {

                console.error(err);

            }

        }
